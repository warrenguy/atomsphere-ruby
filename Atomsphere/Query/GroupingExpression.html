def initialize(params={})
  case params
  when String
  when Symbol
    params = {operator: params}
  end

  params = {
    operator: :and,
    nested_expression: []
  }.merge(params)

  @operator = params[:operator]
  @nested_expression = params[:nested_expression]
end

def to_hash
  {
    expression: {
      operator: operator.to_s,
      nestedExpression: nested_expression.
        map(&:to_hash).
        map{ |h| h[:expression] }
    }
  }
end
